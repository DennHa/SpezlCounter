// Configuration — replace with your Firebase project URL const DB_URL = 'https://spezlcounter-default-rtdb.europe-west1.firebasedatabase.app/match.json'; let lastP1Life = 20; let lastP2Life = 20; // Helper: convert old data to new structure if needed function normalizeData(data) { if (!data) return null; if (data.p1 && data.p1.name !== undefined) { return data; } return null; } // Helper: determine life total color state function getLifeState(life) { if (life <= 0) return 'state-critical'; if (life <= 5) return 'state-danger'; if (life <= 10) return 'state-warning'; return 'state-normal'; } // Update DOM with new life total and animate delta function updateLife(player, newLife) { const id = player === 'p1' ? '#p1-life' : '#p2-life'; const oldLife = player === 'p1' ? lastP1Life : lastP2Life; const element = document.querySelector(id); element.textContent = newLife; element.className = 'life-total ' + getLifeState(newLife); // Show delta animation if changed if (oldLife !== newLife) { const delta = newLife - oldLife; const sign = delta > 0 ? '+' : ''; showDelta(player, sign + delta, element); } if (player === 'p1') { lastP1Life = newLife; } else { lastP2Life = newLife; } } // Display floating delta number function showDelta(player, text, sourceElement) { const deltaEl = document.createElement('div'); deltaEl.className = 'delta ' + (text.startsWith('+') ? 'positive' : 'negative'); deltaEl.textContent = text; deltaEl.style.left = sourceElement.getBoundingClientRect().left + 'px'; deltaEl.style.top = sourceElement.getBoundingClientRect().top + 'px'; document.body.appendChild(deltaEl); setTimeout(() => deltaEl.remove(), 1000); } // Update name in DOM function updateName(player, name) { const id = player === 'p1' ? '#p1-name' : '#p2-name'; document.querySelector(id).textContent = name || 'Spieler ' + (player === 'p1' ? '1' : '2'); } // Poll Firebase Realtime Database async function poll() { try { const response = await fetch(DB_URL + '?t=' + Date.now()); if (!response.ok) throw new Error('Firebase fetch failed'); const data = await response.json(); const normalized = normalizeData(data); if (normalized && normalized.p1 && normalized.p2) { updateName('p1', normalized.p1.name); updateName('p2', normalized.p2.name); updateLife('p1', normalized.p1.life); updateLife('p2', normalized.p2.life); } } catch (error) { console.error('Poll error:', error); } } // Start polling poll(); setInterval(poll, 500);
